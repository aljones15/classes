\begin{thm}
\[
    \frac{2+10-4}{2} = 4.
\]
\end{thm}

\begin{proof}
    We have
\begin{align*}
    \frac{2+10-4}{2}
        &=  \frac{12-4}{2}      \\
        &=  \frac{8}{2}         \\
        &=  4.
\end{align*}
\end{proof}
